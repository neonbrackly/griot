"""
Griot DataFrame Validation - Pandera-Based Validation Engine

This module provides DataFrame validation using Pandera as the underlying
validation engine. It supports multiple DataFrame types:
- pandas DataFrames
- polars DataFrames
- PySpark DataFrames
- Dask DataFrames

The validation engine uses a registry pattern to support different DataFrame
types with their respective validators.

Installation (composable - install only what you need):
    pip install griot-core[pandas]              # pandas support
    pip install griot-core[polars]              # polars support
    pip install griot-core[spark]               # PySpark support
    pip install griot-core[dask]                # Dask support
    pip install griot-core[pandas,polars]       # multiple types
    pip install griot-core[all]                 # all DataFrame types
"""

class DataFrameValidatorRegistry:
    """
    Registry for DataFrame validators.

    Supports automatic validator discovery and registration for
    pandas, polars, pyspark, and dask DataFrames.

    Example:
        # Register a custom validator
        registry = DataFrameValidatorRegistry()
        registry.register(MyCustomValidator())

        # Get validator for a DataFrame
        validator = registry.get_validator(my_dataframe)
        result = validator.validate(my_dataframe, schema)

        # Or use convenience method
        result = registry.validate(my_dataframe, schema)
    """

    _validators: dict[str, DataFrameValidator[Any]]
    _auto_registered: bool



[docs]
    def __init__(self, auto_register: bool = True) -> None:
        """
        Initialize the registry.

        Args:
            auto_register: If True, automatically register available validators
        """
        self._validators = {}
        self._auto_registered = False

        if auto_register:
            self._auto_register_validators()



    def _auto_register_validators(self) -> None:
        """Automatically register validators for installed DataFrame libraries."""
        if self._auto_registered:
            return

        # Try to register pandas validator
        try:
            from griot_core.dataframe_validation import PandasValidator

            self.register(PandasValidator())
        except (ImportError, PanderaNotInstalledError):
            pass

        # Try to register polars validator
        try:
            from griot_core.dataframe_validation import PolarsValidator

            self.register(PolarsValidator())
        except (ImportError, PanderaNotInstalledError):
            pass

        # Try to register pyspark validator
        try:
            from griot_core.dataframe_validation import PySparkValidator

            self.register(PySparkValidator())
        except (ImportError, PanderaNotInstalledError):
            pass

        # Try to register dask validator
        try:
            from griot_core.dataframe_validation import DaskValidator

            self.register(DaskValidator())
        except (ImportError, PanderaNotInstalledError):
            pass

        self._auto_registered = True

    def get_validator(self, df: Any) -> DataFrameValidator[Any]:
        """
        Get the appropriate validator for a DataFrame.

        Args:
            df: DataFrame to find validator for

        Returns:
            DataFrameValidator that can handle the DataFrame

        Raises:
            ValidatorNotFoundError: If no validator found for DataFrame type
        """
        for validator in self._validators.values():
            if validator.is_dataframe_type(df):
                return validator

        # Try to determine type for better error message
        df_type = type(df).__module__.split(".")[0]
        raise ValidatorNotFoundError(df_type)





[docs]
    def get_validator_by_type(self, df_type: DataFrameType | str) -> DataFrameValidator[Any]:
        """
        Get validator by DataFrame type name.

        Args:
            df_type: DataFrame type

        Returns:
            DataFrameValidator for that type

        Raises:
            ValidatorNotFoundError: If no validator registered for type
        """
        key = df_type.value if isinstance(df_type, DataFrameType) else df_type
        if key not in self._validators:
            raise ValidatorNotFoundError(key)
        return self._validators[key]

class PanderaSchemaGenerator:
    """
    Generate Pandera schemas from Schema with ODCS quality rules.

    Converts Schema field definitions to Pandera column schemas,
    transforming ODCS quality rules into Pandera checks.

    Supported ODCS quality types:
        - library: Predefined metrics (nullValues, missingValues, etc.)
        - text: Human-readable only (skipped)
        - sql: SQL queries (skipped - requires DB context)
        - custom: Vendor-specific (skipped)

    Supported library metrics:
        - nullValues: Count of null values
        - missingValues: Count of missing values (null, '', 'N/A', etc.)
        - invalidValues: Count of values not matching pattern/valid set
        - duplicateValues: Count of duplicate values

    Example:
        generator = PanderaSchemaGenerator()
        pandera_schema = generator.generate(griot_schema)
    """

    # Map Griot logical types to Pandera dtypes
    TYPE_MAP: dict[str, str] = {
        "string": "str",
        "integer": "int64",
        "float": "float64",
        "boolean": "bool",
        "date": "datetime64[ns]",
        "datetime": "datetime64[ns]",
        "array": "object",
        "object": "object",
    }



[docs]
    def generate(self, schema: Schema) -> Any:
        """
        Generate a Pandera DataFrameSchema from a Schema.

        Args:
            schema: Schema to convert

        Returns:
            pandera.DataFrameSchema

        Raises:
            PanderaNotInstalledError: If pandera is not installed
        """
        try:
            import pandera as pa
        except ImportError as e:
            raise PanderaNotInstalledError() from e

        columns: dict[str, pa.Column] = {}

        for field_name, field_info in schema.fields.items():
            column = self._create_column(field_info, pa)
            columns[field_name] = column

        # Create schema with optional schema-level checks
        schema_checks = self._create_schema_checks(schema, pa)

        return pa.DataFrameSchema(
            columns=columns,
            checks=schema_checks if schema_checks else None,
            name=schema.name,
            coerce=True,
            strict=False,  # Allow extra columns
        )



    def _create_column(self, field: FieldInfo, pa: Any) -> Any:
        """
        Create a Pandera Column from a FieldInfo.

        Args:
            field: Field metadata with quality rules
            pa: Pandera module

        Returns:
            pa.Column with appropriate checks
        """
        # Determine dtype
        dtype = self.TYPE_MAP.get(field.logical_type.lower(), "object")

        # Build checks from quality rules
        checks: list[Any] = self._process_quality_rules(field.get_dq_checks(), field, pa)

        # Add field-level unique constraint
        if field.unique:
            checks.append(pa.Check(
                lambda s: ~s.duplicated().any(),
                name=f"{field.name}_unique",
                error=f"Column '{field.name}' contains duplicate values"
            ))

        return pa.Column(
            dtype=dtype,
            nullable=field.nullable,
            required=field.required or field.primary_key,
            checks=checks if checks else None,
            name=field.name,
            description=field.description,
        )

    def _process_quality_rules(
        self,
        quality_rules: list[dict[str, Any]],
        field: FieldInfo,
        pa: Any,
    ) -> list[Any]:
        """
        Process ODCS quality rules and convert to Pandera checks.

        Args:
            quality_rules: List of ODCS quality rule dicts
            field: Field info for context
            pa: Pandera module

        Returns:
            List of Pandera Check objects
        """
        checks: list[Any] = []

        for rule in quality_rules:
            rule_type_str = rule.get("type", QualityCheckType.LIBRARY.value)

            # Convert to enum for type safety
            try:
                rule_type = QualityCheckType(rule_type_str)
            except ValueError:
                rule_type = QualityCheckType.LIBRARY if "metric" in rule else None

            if rule_type == QualityCheckType.LIBRARY or "metric" in rule:
                # Library metric - create executable check
                check = self._create_library_check(rule, field, pa)
                if check is not None:
                    checks.append(check)

            elif rule_type == QualityCheckType.TEXT:
                # Text rules are human-readable only, skip
                pass

            elif rule_type == QualityCheckType.SQL:
                # SQL requires database context, skip with note
                # Could store for later execution
                pass

            elif rule_type == QualityCheckType.CUSTOM:
                # Custom rules are vendor-specific, skip
                pass

        return checks

    def _create_library_check(
        self,
        rule: dict[str, Any],
        field: FieldInfo,
        pa: Any,
    ) -> Any | None:
        """
        Create a Pandera check from an ODCS library metric.

        Args:
            rule: Quality rule dictionary
            field: Field info for context
            pa: Pandera module

        Returns:
            pa.Check or None if check cannot be created
        """
        metric_str = rule.get("metric")
        if not metric_str:
            return None

        # Convert metric string to enum for type safety
        try:
            metric = QualityMetric(metric_str)
        except ValueError:
            # Unknown metric, skip
            return None

        # Extract operator and comparison value
        operator = self._extract_operator(rule)
        if operator is None:
            return None

        op_enum, op_value = operator

        # Convert unit string to enum
        unit_str = rule.get("unit", QualityUnit.ROWS.value)
        try:
            unit = QualityUnit(unit_str)
        except ValueError:
            unit = QualityUnit.ROWS

        check_name = rule.get("id") or rule.get("name") or f"{field.name}_{metric.value}"
        arguments = rule.get("arguments", {})

        # Dispatch to metric-specific handler based on enum
        if metric == QualityMetric.NULL_VALUES:
            return self._create_null_check(op_enum, op_value, unit, check_name, pa)

        elif metric == QualityMetric.MISSING_VALUES:
            missing_values = arguments.get("missingValues", [None, "", "N/A", "n/a"])
            return self._create_missing_check(missing_values, op_enum, op_value, unit, check_name, pa)

        elif metric == QualityMetric.INVALID_VALUES:
            return self._create_invalid_values_check(arguments, rule, op_enum, op_value, unit, check_name, field, pa)

        elif metric == QualityMetric.DUPLICATE_VALUES:
            # Property-level duplicate check (not schema-level)
            if "properties" not in arguments:
                return self._create_duplicate_check(op_enum, op_value, unit, check_name, pa)

        return None

    def _extract_operator(self, rule: dict[str, Any]) -> tuple[QualityOperator, Any] | None:
        """
        Extract operator enum and comparison value from a quality rule.

        Args:
            rule: Quality rule dictionary

        Returns:
            Tuple of (QualityOperator, comparison_value) or None
        """
        for op in QualityOperator:
            if op.value in rule:
                return (op, rule[op.value])
        return None

    def _create_null_check(
        self,
        op: QualityOperator,
        op_value: Any,
        unit: QualityUnit,
        check_name: str,
        pa: Any,
    ) -> Any:
        """
        Create a null values check.

        Counts null values in the column and compares against threshold.
        Supports both row count and percentage units.

        Args:
            op: Comparison operator enum
            op_value: Threshold value
            unit: Unit enum (ROWS or PERCENT)
            check_name: Name for the check
            pa: Pandera module

        Returns:
            pa.Check for null values
        """
        # Factory function to properly capture closure variables
        def make_check(op: QualityOperator, op_value: Any, unit: QualityUnit):
            def check_fn(series):
                null_count = series.isna().sum()
                total = len(series)
                metric_value = unit.calculate_metric(null_count, total)
                return op.compare(metric_value, op_value)
            return check_fn

        return pa.Check(
            make_check(op, op_value, unit),
            name=check_name,
            error=f"Null values check failed: expected {op.value} {op_value} ({unit.value})"
        )

    def _create_missing_check(
        self,
        missing_values: list[Any],
        op: QualityOperator,
        op_value: Any,
        unit: QualityUnit,
        check_name: str,
        pa: Any,
    ) -> Any:
        """
        Create a missing values check.

        Counts values considered as missing (null, empty string, N/A, etc.)
        and compares against threshold.

        Args:
            missing_values: List of values considered missing
            op: Comparison operator enum
            op_value: Threshold value
            unit: Unit enum (ROWS or PERCENT)
            check_name: Name for the check
            pa: Pandera module

        Returns:
            pa.Check for missing values
        """
        def make_check(missing_values: list[Any], op: QualityOperator, op_value: Any, unit: QualityUnit):
            # Separate None from other missing values for proper handling
            non_null_missing = [v for v in missing_values if v is not None]
            check_null = None in missing_values

            def check_fn(series):
                # Count nulls if None is in missing values
                if check_null:
                    is_missing = series.isna()
                else:
                    is_missing = series.apply(lambda x: False)

                # Count other missing values
                if non_null_missing:
                    is_missing = is_missing | series.isin(non_null_missing)

                missing_count = is_missing.sum()
                total = len(series)
                metric_value = unit.calculate_metric(missing_count, total)
                return op.compare(metric_value, op_value)
            return check_fn

        return pa.Check(
            make_check(missing_values, op, op_value, unit),
            name=check_name,
            error=f"Missing values check failed: expected {op.value} {op_value} ({unit.value})"
        )

    def _create_invalid_values_check(
        self,
        arguments: dict[str, Any],
        rule: dict[str, Any],
        op: QualityOperator,
        op_value: Any,
        unit: QualityUnit,
        check_name: str,
        field: FieldInfo,
        pa: Any,
    ) -> Any | None:
        """
        Create an invalid values check.

        Supports two modes:
        1. Pattern matching: Values must match a regex pattern
        2. Valid values list: Values must be in a predefined set

        Args:
            arguments: Rule arguments (validValues, pattern)
            rule: Full rule dict (pattern can be at top level)
            op: Comparison operator enum
            op_value: Threshold value
            unit: Unit enum (ROWS or PERCENT)
            check_name: Name for the check
            field: Field info
            pa: Pandera module

        Returns:
            pa.Check for invalid values or None
        """
        pattern = arguments.get("pattern") or rule.get("pattern")
        valid_values = arguments.get("validValues")

        if pattern:
            # Pattern-based validation
            if op == QualityOperator.MUST_BE and op_value == 0:
                # Simple case: all values must match pattern
                return pa.Check.str_matches(
                    pattern,
                    name=check_name,
                    error=f"Values must match pattern: {pattern}"
                )
            else:
                # Count non-matching values and compare
                def make_pattern_check(pattern: str, op: QualityOperator, op_value: Any, unit: QualityUnit):
                    import re
                    compiled = re.compile(pattern)

                    def check_fn(series):
                        # Count values that don't match the pattern
                        def matches(val):
                            if val is None or (isinstance(val, float) and val != val):  # NaN check
                                return True  # Nulls handled separately
                            return bool(compiled.match(str(val)))

                        invalid_count = (~series.apply(matches)).sum()
                        total = len(series)
                        metric_value = unit.calculate_metric(invalid_count, total)
                        return op.compare(metric_value, op_value)
                    return check_fn

                return pa.Check(
                    make_pattern_check(pattern, op, op_value, unit),
                    name=check_name,
                    error=f"Invalid values check failed: expected {op.value} {op_value} ({unit.value})"
                )

        elif valid_values:
            # Valid values list validation
            if op == QualityOperator.MUST_BE and op_value == 0:
                # Simple case: all values must be in valid set
                return pa.Check.isin(
                    valid_values,
                    name=check_name,
                    error=f"Values must be one of: {valid_values}"
                )
            else:
                # Count invalid values and compare
                def make_valid_values_check(valid_values: list, op: QualityOperator, op_value: Any, unit: QualityUnit):
                    def check_fn(series):
                        # Count values not in valid set (excluding nulls)
                        non_null = series.dropna()
                        invalid_count = (~non_null.isin(valid_values)).sum()
                        total = len(series)
                        metric_value = unit.calculate_metric(invalid_count, total)
                        return op.compare(metric_value, op_value)
                    return check_fn

                return pa.Check(
                    make_valid_values_check(valid_values, op, op_value, unit),
                    name=check_name,
                    error=f"Invalid values check failed: expected {op.value} {op_value} ({unit.value})"
                )

        return None

    def _create_duplicate_check(
        self,
        op: QualityOperator,
        op_value: Any,
        unit: QualityUnit,
        check_name: str,
        pa: Any,
    ) -> Any:
        """
        Create a duplicate values check for a single column.

        Counts duplicate values and compares against threshold.

        Args:
            op: Comparison operator enum
            op_value: Threshold value
            unit: Unit enum (ROWS or PERCENT)
            check_name: Name for the check
            pa: Pandera module

        Returns:
            pa.Check for duplicate values
        """
        def make_check(op: QualityOperator, op_value: Any, unit: QualityUnit):
            def check_fn(series):
                # Count duplicate values (not including first occurrence)
                dup_count = series.duplicated().sum()
                total = len(series)
                metric_value = unit.calculate_metric(dup_count, total)
                return op.compare(metric_value, op_value)
            return check_fn

        return pa.Check(
            make_check(op, op_value, unit),
            name=check_name,
            error=f"Duplicate values check failed: expected {op.value} {op_value} ({unit.value})"
        )

    def _create_schema_checks(self, schema: Schema, pa: Any) -> list[Any]:
        """
        Create schema-level checks from quality rules.

        Handles schema-level metrics from schema.quality:
        - ROW_COUNT: Total row count validation
        - DUPLICATE_ROWS: Multi-column uniqueness (duplicateValues with properties)

        Args:
            schema: Schema with quality rules
            pa: Pandera module

        Returns:
            List of schema-level pa.Check objects
        """
        checks: list[Any] = []

        # Get schema-level quality rules (from schema.quality, not field.quality)
        schema_quality = getattr(schema, "quality", []) or []
        # Also check class variable for class-based schemas
        if not schema_quality and hasattr(schema, "_quality"):
            schema_quality = schema._quality or []

        for rule in schema_quality:
            rule_type_str = rule.get("type", QualityCheckType.LIBRARY.value)

            # Only process library-type rules
            try:
                rule_type = QualityCheckType(rule_type_str)
            except ValueError:
                rule_type = QualityCheckType.LIBRARY if "metric" in rule else None

            if rule_type != QualityCheckType.LIBRARY and "metric" not in rule:
                continue

            metric_str = rule.get("metric")
            if not metric_str:
                continue

            # Convert metric to enum
            try:
                metric = QualityMetric(metric_str)
            except ValueError:
                continue

            # Only process schema-level metrics
            if metric.is_property_level and "properties" not in rule.get("arguments", {}):
                continue

            operator = self._extract_operator(rule)
            if operator is None:
                continue

            op, op_value = operator
            check_name = rule.get("id") or rule.get("name") or f"schema_{metric.value}"
            arguments = rule.get("arguments", {})

            # Convert unit to enum
            unit_str = rule.get("unit", QualityUnit.ROWS.value)
            try:
                unit = QualityUnit(unit_str)
            except ValueError:
                unit = QualityUnit.ROWS

            if metric == QualityMetric.ROW_COUNT:
                # Row count check (schema-level)
                def make_row_count_check(op: QualityOperator, op_value: Any):
                    def check_fn(df):
                        row_count = len(df)
                        return op.compare(row_count, op_value)
                    return check_fn

                checks.append(pa.Check(
                    make_row_count_check(op, op_value),
                    name=check_name,
                    error=f"Row count check failed: expected {op.value} {op_value}"
                ))

            elif metric == QualityMetric.DUPLICATE_VALUES and "properties" in arguments:
                # Multi-column duplicate check (schema-level)
                columns = arguments["properties"]

                def make_multi_dup_check(columns: list, op: QualityOperator, op_value: Any, unit: QualityUnit):
                    def check_fn(df):
                        # Check for duplicates across specified columns
                        dup_count = df.duplicated(subset=columns).sum()
                        total = len(df)
                        metric_value = unit.calculate_metric(dup_count, total)
                        return op.compare(metric_value, op_value)
                    return check_fn

                checks.append(pa.Check(
                    make_multi_dup_check(columns, op, op_value, unit),
                    name=check_name,
                    error=f"Multi-column duplicate check failed on {columns}: expected {op.value} {op_value}"
                ))

            elif metric == QualityMetric.DUPLICATE_ROWS:
                # Duplicate rows check (all columns or specified columns)
                columns = arguments.get("properties")  # Optional - if None, checks all columns

                def make_dup_rows_check(columns: list | None, op: QualityOperator, op_value: Any, unit: QualityUnit):
                    def check_fn(df):
                        dup_count = df.duplicated(subset=columns).sum()
                        total = len(df)
                        metric_value = unit.calculate_metric(dup_count, total)
                        return op.compare(metric_value, op_value)
                    return check_fn

                checks.append(pa.Check(
                    make_dup_rows_check(columns, op, op_value, unit),
                    name=check_name,
                    error=f"Duplicate rows check failed: expected {op.value} {op_value} ({unit.value})"
                ))

        return checks

    def validate(
        self,
        df: Any,
        schema: Schema,
        *,
        mode: ValidationMode = ValidationMode.LAZY,
    ) -> DataFrameValidationResult:
        """Validate a pandas DataFrame against a schema."""
        import time

        import pandera as pa

        start_time = time.perf_counter()

        # Generate pandera schema
        pandera_schema = self._schema_generator.generate(schema)

        # Validate
        errors: list[dict[str, Any]] = []
        field_results: dict[str, FieldValidationResult] = {}
        is_valid = True

        try:
            if mode == ValidationMode.LAZY:
                pandera_schema.validate(df, lazy=True)
            else:
                pandera_schema.validate(df, lazy=False)
        except pa.errors.SchemaErrors as e:
            is_valid = False
            # Process schema errors
            for _, row in e.failure_cases.iterrows():
                error_dict = {
                    "field": row.get("column"),
                    "check": row.get("check"),
                    "row": row.get("index"),
                    "value": row.get("failure_case"),
                }
                errors.append(error_dict)

                # Update field results
                field_name = row.get("column", "unknown")
                if field_name not in field_results:
                    field_results[field_name] = FieldValidationResult(
                        field_name=field_name,
                        is_valid=False,
                        error_count=0,
                        errors=[],
                    )
                field_results[field_name].error_count += 1
                field_results[field_name].errors.append(error_dict)

        except pa.errors.SchemaError as e:
            is_valid = False
            errors.append({
                "field": str(e.schema),
                "check": str(e.check),
                "message": str(e),
            })

        duration_ms = (time.perf_counter() - start_time) * 1000

        return DataFrameValidationResult(
            is_valid=is_valid,
            schema_name=schema.name,
            row_count=len(df),
            error_count=len(errors),
            field_results=field_results,
            errors=errors,
            duration_ms=duration_ms,
            dataframe_type=self.dataframe_type.value,
        )

        "polars": ("polars", "pandera.polars"),
        "spark": ("pyspark", "pandera.pyspark"),
        "pyspark": ("pyspark", "pandera.pyspark"),
        "dask": ("dask.dataframe", "pandera"),
    }

    if backend.lower() not in extra_map:
        return {
            "available": False,
            "version": None,
            "error": f"Unknown backend: {backend}",
            "install": None,
        }

    lib, pandera_module = extra_map[backend.lower()]
    extra_name = "spark" if backend.lower() in ("spark", "pyspark") else backend.lower()

    try:
        import importlib

        # Check DataFrame library
        df_lib = importlib.import_module(lib.split(".")[0])
        version = getattr(df_lib, "__version__", "unknown")

        # Check pandera module
        importlib.import_module(pandera_module.split(".")[0])

        return {
            "available": True,
            "version": version,
            "install": f"pip install griot-core[{extra_name}]",
        }
    except ImportError:
        return {
            "available": False,
            "version": None,
            "install": f"pip install griot-core[{extra_name}]",
        }

def get_available_backends() -> dict[str, dict[str, Any]]:
    """
    Get information about all available DataFrame backends.

    Returns:
        Dict mapping backend name to availability info

    Example:
        >>> get_available_backends()
        {
            'pandas': {'available': True, 'version': '2.0.0', ...},
            'polars': {'available': False, 'install': 'pip install griot-core[polars]'},
            ...
        }
    """
    backends = ["pandas", "polars", "spark", "dask"]
    return {backend: check_backend(backend) for backend in backends}
